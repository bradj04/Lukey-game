        let audioCtx = null;

        function initAudio() {
            if (!audioCtx) {
                audioCtx = new (window.AudioContext || window.webkitAudioContext)();
            }
        }

        function playSound(type) {
            if (!audioCtx) return;
            if (audioCtx.state === 'suspended') audioCtx.resume();
            const now = audioCtx.currentTime;

            if (type === 'pop') {
                let osc = audioCtx.createOscillator();
                let gain = audioCtx.createGain();
                osc.type = 'sine';
                osc.frequency.setValueAtTime(400, now);
                osc.frequency.exponentialRampToValueAtTime(80, now + 0.08);
                gain.gain.setValueAtTime(0.4, now);
                gain.gain.linearRampToValueAtTime(0.01, now + 0.08);
                osc.connect(gain);
                gain.connect(audioCtx.destination);
                osc.start(now); osc.stop(now + 0.08);
            } 
            else if (type === 'slice') {
                let bufferSize = audioCtx.sampleRate * 0.08;
                let buffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate);
                let data = buffer.getChannelData(0);
                for (let i = 0; i < bufferSize; i++) data[i] = Math.random() * 2 - 1;
                let noise = audioCtx.createBufferSource();
                noise.buffer = buffer;
                let filter = audioCtx.createBiquadFilter();
                filter.type = 'bandpass';
                filter.frequency.setValueAtTime(1200, now);
                let gain = audioCtx.createGain();
                gain.gain.setValueAtTime(0.3, now);
                gain.gain.linearRampToValueAtTime(0.01, now + 0.08);
                noise.connect(filter); filter.connect(gain); gain.connect(audioCtx.destination);
                noise.start(now); noise.stop(now + 0.08);
            } 
            else if (type === 'crack') {
                let osc = audioCtx.createOscillator();
                let gain = audioCtx.createGain();
                osc.type = 'triangle';
                osc.frequency.setValueAtTime(160, now);
                gain.gain.setValueAtTime(0.5, now);
                gain.gain.linearRampToValueAtTime(0.01, now + 0.05);
                osc.connect(gain); gain.connect(audioCtx.destination);
                osc.start(now); osc.stop(now + 0.05);
            } 
            else if (type === 'roar') {
                [140, 180, 240].forEach((freq, idx) => {
                    let osc = audioCtx.createOscillator();
                    let gain = audioCtx.createGain();
                    osc.type = 'sawtooth';
                    osc.frequency.setValueAtTime(freq, now);
                    osc.frequency.linearRampToValueAtTime(freq * 0.8, now + 0.5);
                    let filter = audioCtx.createBiquadFilter();
                    filter.type = 'lowpass';
                    filter.frequency.setValueAtTime(700, now);
                    gain.gain.setValueAtTime(0.2, now);
                    gain.gain.linearRampToValueAtTime(0.01, now + 0.5);
                    osc.connect(filter); filter.connect(gain); gain.connect(audioCtx.destination);
                    osc.start(now + idx * 0.01); osc.stop(now + 0.5);
                });
            }
        }

        function startApp() {
            initAudio();
            document.getElementById('splash-screen').style.display = 'none';
        }

        let currentMode = null;
        let animationFrameId = null;

        const balloonCanvas = document.getElementById('balloonCanvas');
        const fruitCanvas = document.getElementById('fruitCanvas');
        const dinoCanvas = document.getElementById('dinoCanvas');

        function resizeCanvases() {
            [balloonCanvas, fruitCanvas, dinoCanvas].forEach(c => {
                c.width = window.innerWidth;
                c.height = window.innerHeight;
            });
        }

            initAudio();

            else if (mode === 'dino') { document.getElementById('dino-screen').style.display = 'block'; startDino(); }
        }

        function goHome() {
            if (animationFrameId) cancelAnimationFrame(animationFrameId);
            document.getElementById('balloon-screen').style.display = 'none';
            document.getElementById('fruit-screen').style.display = 'none';

        // --- BALLOONS ENGINE ---
        let balloons = [];
        let popParticles = [];
        const balloonColors = ['#ff4d6d', '#ff758f', '#3a86ff', '#8338ec', '#ff006e', '#06d6a0', '#ffbe0b', '#fb5607'];

        function createBalloon() {
            return {
                x: Math.random() * (balloonCanvas.width - 80) + 40,
                y: balloonCanvas.height + 60,
                radius: Math.random() * 15 + 35,
                color: balloonColors[Math.floor(Math.random() * balloonColors.length)],
                speed: Math.random() * 1.5 + 1.5,
                swaySpeed: Math.random() * 0.02 + 0.01,
                swayAmount: Math.random() * 15 + 5,
                seed: Math.random() * 100
            };
        }

        function drawBalloonGraphic(ctx, x, y, r, color) {
            ctx.save();
            ctx.beginPath();
            ctx.moveTo(x, y + r);
            ctx.bezierCurveTo(x - 5, y + r + 15, x + 5, y + r + 30, x, y + r + 45);
            ctx.strokeStyle = '#94a3b8';
            ctx.lineWidth = 2;
            ctx.stroke();

            ctx.beginPath();
            ctx.moveTo(x, y + r - 2);
            ctx.lineTo(x - 6, y + r + 8);
            ctx.lineTo(x + 6, y + r + 8);
            ctx.closePath();
            ctx.fillStyle = color;
            ctx.fill();

            ctx.beginPath();
            ctx.ellipse(x, y, r * 0.85, r, 0, 0, Math.PI * 2);
            let gradient = ctx.createRadialGradient(x - r*0.2, y - r*0.3, r*0.1, x, y, r);
            gradient.addColorStop(0, '#ffffff');
            gradient.addColorStop(0.2, color);
            gradient.addColorStop(1, '#00000099');
            ctx.fillStyle = gradient;
            ctx.fill();
            ctx.restore();
        }

        function startBalloons() {
            balloons = []; popParticles = [];
            for(let i=0; i<5; i++) { balloons.push(createBalloon()); balloons[i].y = Math.random() * balloonCanvas.height; }
            const ctx = balloonCanvas.getContext('2d');
            function update() {
                ctx.clearRect(0, 0, balloonCanvas.width, balloonCanvas.height);
                if (balloons.length < 6 && Math.random() < 0.02) balloons.push(createBalloon());
                for (let i = balloons.length - 1; i >= 0; i--) {
                    let b = balloons[i]; b.y -= b.speed;
                    let sway = Math.sin(b.seed + b.y * b.swaySpeed) * b.swayAmount;
                    drawBalloonGraphic(ctx, b.x + sway, b.y, b.radius, b.color);
                    if (b.y < -b.radius * 2) balloons.splice(i, 1);
                }
                for (let i = popParticles.length - 1; i >= 0; i--) {
                    let p = popParticles[i]; p.x += p.vx; p.y += p.vy; p.alpha -= 0.05;
                    ctx.save(); ctx.globalAlpha = Math.max(0, p.alpha); ctx.beginPath();
                    ctx.arc(p.x, p.y, p.size, 0, Math.PI*2); ctx.fillStyle = p.color; ctx.fill(); ctx.restore();
                    if (p.alpha <= 0) popParticles.splice(i, 1);
                }
                animationFrameId = requestAnimationFrame(update);
            }
            update();
        }

        function handleBalloonTouch(e) {
            let clickX = e.clientX || (e.touches && e.touches[0].clientX);
            let clickY = e.clientY || (e.touches && e.touches[0].clientY);
            if (!clickX) return;
            for (let i = balloons.length - 1; i >= 0; i--) {
                let b = balloons[i];
                let sway = Math.sin(b.seed + b.y * b.swaySpeed) * b.swayAmount;
                if (Math.hypot(clickX - (b.x + sway), clickY - b.y) < b.radius + 15) {
                    playSound('pop');
                    for (let p = 0; p < 12; p++) {
                        let angle = Math.random() * Math.PI * 2; let speed = Math.random() * 4 + 2;
                        popParticles.push({ x: b.x + sway, y: b.y, vx: Math.cos(angle) * speed, vy: Math.sin(angle) * speed, size: Math.random() * 4 + 3, color: b.color, alpha: 1 });
                    }
                    balloons.splice(i, 1); break;
                }
            }
        }
        balloonCanvas.addEventListener('touchstart', function(e){ e.preventDefault(); handleBalloonTouch(e); });
        balloonCanvas.addEventListener('mousedown', handleBalloonTouch);

        // --- FRUIT ENGINE ---
        let fruits = []; let fruitJuice = [];
        const fruitEmojis = [
            { char: '🍉', color: '#ff3366' }, { char: '🍎', color: '#ff3b30' },
            { char: '🍓', color: '#ff2d55' }, { char: ' Pineapple ', color: '#ffcc00', char: '🍍' },
            { char: '🍊', color: '#ff9500' }, { char: '🍌', color: '#ffcc00' }
        ];

        function createFruit() {
            let em = fruitEmojis[Math.floor(Math.random() * fruitEmojis.length)];
            return { x: Math.random() * (fruitCanvas.width - 100) + 50, y: fruitCanvas.height + 50, vx: (Math.random() - 0.5) * 4, vy: -(Math.random() * 4 + 11), gravity: 0.22, emoji: em.char, color: em.color, size: 65, sliced: false, halfX1: 0, halfY1: 0, halfX2: 0, halfY2: 0 };
        }

        function startFruit() {
            fruits = []; fruitJuice = [];
            const ctx = fruitCanvas.getContext('2d');
            function update() {
                ctx.clearRect(0, 0, fruitCanvas.width, fruitCanvas.height);
                if (fruits.length < 4 && Math.random() < 0.025) fruits.push(createFruit());
                for (let i = fruits.length - 1; i >= 0; i--) {
                    let f = fruits[i]; f.vy += f.gravity; f.x += f.vx; f.y += f.vy;
                    ctx.save(); ctx.font = `${f.size}px Arial`; ctx.textAlign = 'center'; ctx.textBaseline = 'middle';
                    if (!f.sliced) {
                        ctx.translate(f.x, f.y); ctx.fillText(f.emoji, 0, 0);
                    } else {
                        f.halfX1 -= 2; f.halfY1 += 1; f.halfX2 += 2; f.halfY2 += 1;
                        ctx.save(); ctx.translate(f.x + f.halfX1, f.y + f.halfY1); ctx.beginPath(); ctx.rect(-f.size, -f.size, f.size, f.size*2); ctx.clip(); ctx.fillText(f.emoji, 0, 0); ctx.restore();
                        ctx.save(); ctx.translate(f.x + f.halfX2, f.y + f.halfY2); ctx.beginPath(); ctx.rect(0, -f.size, f.size, f.size*2); ctx.clip(); ctx.fillText(f.emoji, 0, 0); ctx.restore();
                    }
                    ctx.restore();
                    if (f.y > fruitCanvas.height + 150) fruits.splice(i, 1);
                }
                for (let i = fruitJuice.length - 1; i >= 0; i--) {
                    let j = fruitJuice[i]; j.x += j.vx; j.y += j.vy; j.alpha -= 0.04;
                    ctx.save(); ctx.globalAlpha = Math.max(0, j.alpha); ctx.beginPath(); ctx.arc(j.x, j.y, j.r, 0, Math.PI*2); ctx.fillStyle = j.color; ctx.fill(); ctx.restore();
                    if (j.alpha <= 0) fruitJuice.splice(i, 1);
                }
                animationFrameId = requestAnimationFrame(update);
            }
            update();
        }

        function checkFruitSlice(e) {
            let x = e.clientX || (e.touches && e.touches[0].clientX);
            let y = e.clientY || (e.touches && e.touches[0].clientY);
            if (!x || !y) return;
            for (let f of fruits) {
                if (!f.sliced && Math.hypot(x - f.x, y - f.y) < f.size * 0.8) {
                    f.sliced = true; playSound('slice');
                    for (let j = 0; j < 15; j++) {
                        let angle = Math.random() * Math.PI * 2; let spd = Math.random() * 5 + 2;
                        fruitJuice.push({ x: f.x, y: f.y, vx: Math.cos(angle) * spd, vy: Math.sin(angle) * spd, r: Math.random() * 6 + 4, color: f.color, alpha: 1 });
                    }
                }
            }
        }
        fruitCanvas.addEventListener('touchmove', function(e){ checkFruitSlice(e); });
        fruitCanvas.addEventListener('touchstart', function(e){ checkFruitSlice(e); });
        fruitCanvas.addEventListener('mousemove', checkFruitSlice);
        fruitCanvas.addEventListener('mousedown', checkFruitSlice);

        // --- DINOSAUR ENGINE (3 Species + Fire Roast Effect) ---
        let eggTaps = 0;
        let isHatched = false;
        let eggShake = 0;
        let dinoScale = 0;
        let fireParticles = [];
        let dinoIndex = 0; // 0 = Brachiosaurus, 1 = Triceratops, 2 = T-Rex
        
        const dinoTypes = [
            { name: "Brachiosaurus", color: "#22c55e", type: "longneck" },
            { name: "Triceratops", color: "#3a86ff", type: "tricera" },
            { name: "T-Rex", color: "#ef4444", type: "trex" }
        ];

        function startDino() {
            eggTaps = 0; isHatched = false; eggShake = 0; dinoScale = 0; fireParticles = [];
            const ctx = dinoCanvas.getContext('2d');

            function update() {
                ctx.clearRect(0, 0, dinoCanvas.width, dinoCanvas.height);
                let midX = dinoCanvas.width / 2;
                let midY = dinoCanvas.height / 2 + 50;
                let currentDino = dinoTypes[dinoIndex];

                if (!isHatched) {
                    ctx.save();
                    ctx.translate(midX, midY);
                    if (eggShake > 0) { ctx.rotate(Math.sin(Date.now() * 0.05) * 0.08); eggShake--; }

                    ctx.beginPath();
                    ctx.moveTo(0, -110); ctx.bezierCurveTo(75, -110, 85, 70, 0, 80); ctx.bezierCurveTo(-85, 70, -75, -110, 0, -110); ctx.closePath();
                    let grad = ctx.createRadialGradient(-15, -30, 10, 0, 0, 100);
                    grad.addColorStop(0, '#fffbeb'); grad.addColorStop(0.8, '#fef08a'); grad.addColorStop(1, '#facc15');
                    ctx.fillStyle = grad; ctx.fill(); ctx.strokeStyle = '#ca8a04'; ctx.lineWidth = 4; ctx.stroke();

                    ctx.fillStyle = '#fde047';
                    ctx.beginPath(); ctx.arc(-30, 10, 12, 0, Math.PI*2); ctx.fill();
                    ctx.beginPath(); ctx.arc(35, -20, 15, 0, Math.PI*2); ctx.fill();

                    if (eggTaps > 0) {
                        ctx.beginPath(); ctx.moveTo(-15, -30); ctx.lineTo(5, -10); ctx.lineTo(-10, 10);
                        if (eggTaps > 1) { ctx.moveTo(5, -10); ctx.lineTo(35, -5); ctx.lineTo(20, 25); }
                        ctx.strokeStyle = '#ca8a04'; ctx.lineWidth = 4; ctx.stroke();
                    }
                    ctx.restore();

                    ctx.font = 'bold 22px Arial'; ctx.fillStyle = '#475569'; ctx.textAlign = 'center';
                    ctx.fillText('Tap the egg to hatch it!', midX, midY - 180);
                } else {
                    if (dinoScale < 1) dinoScale += 0.08;

                    ctx.save();
                    ctx.translate(midX, midY);
                    ctx.scale(dinoScale, dinoScale);

                    // Render current selected dinosaur species vectors
                    ctx.fillStyle = currentDino.color;
                    
                    if (currentDino.type === "longneck") {
                        // Brachiosaurus
                        ctx.beginPath(); ctx.moveTo(-40, 30); ctx.quadraticCurveTo(-110, 40, -100, -10); ctx.quadraticCurveTo(-80, 0, -30, 20); ctx.fill(); // Tail
                        ctx.fillRect(-25, 60, 22, 25); ctx.fillRect(10, 60, 22, 25); // Legs
                        ctx.beginPath(); ctx.arc(0, 20, 55, 0, Math.PI * 2); ctx.fill(); // Body
                        ctx.beginPath(); ctx.moveTo(20, 0); ctx.quadraticCurveTo(60, -40, 50, -80); ctx.quadraticCurveTo(85, -120, 40, -120); ctx.quadraticCurveTo(10, -110, 10, -70); ctx.closePath(); ctx.fill(); // Head
                    } 
                    else if (currentDino.type === "tricera") {
                        // Triceratops
                        ctx.beginPath(); ctx.moveTo(-45, 30); ctx.quadraticCurveTo(-95, 20, -90, -5); ctx.quadraticCurveTo(-70, -5, -35, 20); ctx.fill(); // Tail
                        ctx.fillRect(-35, 50, 22, 30); ctx.fillRect(15, 50, 22, 30); // Legs
                        ctx.beginPath(); ctx.arc(-5, 15, 55, 0, Math.PI * 2); ctx.fill(); // Body
                        ctx.beginPath(); ctx.arc(45, -5, 30, 0, Math.PI*2); ctx.fill(); // Head
                        ctx.fillStyle = '#fef08a'; // Horns/Frill Spikes
                        ctx.beginPath(); ctx.moveTo(40, -30); ctx.lineTo(45, -55); ctx.lineTo(55, -32); ctx.fill(); 
                        ctx.beginPath(); ctx.moveTo(55, -20); ctx.lineTo(80, -32); ctx.lineTo(65, -5); ctx.fill();
                    } 
                    else if (currentDino.type === "trex") {
                        // T-Rex (Big Jaw, Tiny Arms)
                        ctx.beginPath(); ctx.moveTo(-40, 30); ctx.quadraticCurveTo(-110, 60, -100, 10); ctx.quadraticCurveTo(-80, 10, -30, 30); ctx.fill(); // Tail
                        ctx.fillRect(-20, 55, 25, 30); ctx.fillRect(5, 55, 25, 30); // Heavy Legs
                        ctx.beginPath(); ctx.arc(-5, 15, 55, 0, Math.PI * 2); ctx.fill(); // Torso
                        ctx.beginPath(); ctx.rect(10, -60, 65, 55); ctx.fill(); // Giant Skull Box
                        ctx.beginPath(); ctx.rect(10, -5, 50, 20); ctx.fill(); // Lower Jaw
                        ctx.fillStyle = '#ffffff'; ctx.fillRect(20, -10, 8, 8); ctx.fillRect(35, -10, 8, 8); // Sharp Teeth
                        ctx.fillStyle = currentDino.color; ctx.fillRect(20, 15, 15, 8); // Tiny arms
                    }

                    // Shared Friendly Eye Features
                    ctx.fillStyle = '#ffffff'; ctx.beginPath(); ctx.arc(38, -100, 7, 0, Math.PI*2); ctx.fill();
                    ctx.fillStyle = '#000000'; ctx.beginPath(); ctx.arc(40, -100, 3.5, 0, Math.PI*2); ctx.fill();
                    ctx.fillStyle = '#f43f5e'; ctx.save(); ctx.globalAlpha = 0.5; ctx.beginPath(); ctx.arc(48, -85, 8, 0, Math.PI*2); ctx.fill(); ctx.restore();

                    ctx.restore();

                    // Process & Draw Roasting Fire Particles
                    for (let i = fireParticles.length - 1; i >= 0; i--) {
                        let p = fireParticles[i]; p.x += p.vx; p.y += p.vy; p.size *= 0.96; p.alpha -= 0.02;
                        ctx.save(); ctx.globalAlpha = Math.max(0, p.alpha); ctx.beginPath();
                        ctx.arc(p.x, p.y, p.size, 0, Math.PI*2);
                        ctx.fillStyle = p.color; ctx.fill(); ctx.restore();
                        if (p.alpha <= 0 || p.size < 2) fireParticles.splice(i, 1);
                    }

                    ctx.font = 'bold 28px Arial'; ctx.fillStyle = currentDino.color; ctx.textAlign = 'center';
                    ctx.fillText(`RAWR! It's a ${currentDino.name}! 🔥`, midX, midY - 180);
                    ctx.font = '16px Arial'; ctx.fillStyle = '#64748b';
                    ctx.fillText('(Tap him to roast fire & change dino!)', midX, midY - 150);
                }
                animationFrameId = requestAnimationFrame(update);
            }
            update();
        }

        function handleDinoTouch(e) {
            e.preventDefault();
            let midX = dinoCanvas.width / 2;
            let midY = dinoCanvas.height / 2 + 50;

            if (!isHatched) {
                eggTaps++; eggShake = 20;
                if (eggTaps >= 3) { isHatched = true; playSound('roar'); } else { playSound('crack'); }
            } else {
                playSound('roar');
                dinoScale = 0.7; // bounce effect

                // Trigger massive cartoon flame roast bursts from mouth position area
                for (let i = 0; i < 25; i++) {
                    let angle = (Math.random() - 0.5) * 0.6; // focused forward stream
                    let speed = Math.random() * 8 + 4;
                    let colors = ['#ef4444', '#f97316', '#facc15', '#ffedd5'];
                    fireParticles.push({
                        x: midX + 60, y: midY - 60,
                        vx: Math.cos(angle) * speed,
                        vy: Math.sin(angle) * speed - (Math.random() * 2), // float slightly upwards
                        size: Math.random() * 15 + 15,
                        color: colors[Math.floor(Math.random() * colors.length)],
                        alpha: 1.0
                    });
                }

                // Cycle to next dinosaur type after a mini timeout delay so he can view the flame spray
                setTimeout(() => {
                    eggTaps = 0;
                    isHatched = false; // put it back in the egg to crack again!
                    dinoIndex = (dinoIndex + 1) % dinoTypes.length;
                }, 900);
            }
        }
        dinoCanvas.addEventListener('touchstart', handleDinoTouch);
        dinoCanvas.addEventListener('mousedown', handleDinoTouch);
